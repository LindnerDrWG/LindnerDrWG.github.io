--................................. Prerequisites: sort, Rank etc.
min(a,b) = test(a<=b, a,b)

median(X, xmed,Xs,n2,n2p) = do(
          Xs   = sort(X),
          n2   = floor(dim(X)/2),  
          n2p  = n2+1,
          xmed = test( not(mod(dim(X),2)=0), Xs[n2p], 
                       0.5*(Xs[n2]+Xs[n2p])), float(xmed))

stdnormCDF(x)  = 1/2*erfc(-x/sqrt(2))

sort(v,i,k,n,t) = do( n = dim(v),     -- by G. Weigt, 19.7.2025
     loop( test(n < 2, break), k = 1,
           for(i, 1, n - 1,
               test(v[i] > v[i + 1], 
                 do( t = v[i], v[i] = v[i + 1], v[i + 1] = t,
                     k = i ))), n = k), v)

Rank(W, i,j,k,n,t, Srank) = do(       -- by G. Weigt 8/25  
         j = 1, n = dim(W), Srank = 0,
         loop( test(j > n, break),
               k = j + 1,
               loop( test(k > n, break, W[j]==W[k], k=k+1, break)),
               t = 0.5*(j + k - 1),
               for(i, j, k-1, W[i] = t),
               t = k - j,
               Srank = Srank + t^3 - t,
               j = k), Srank,W)

correspond(X,sX, M) = do(   -- X and sorted X, by W. Lindner '25
    M=zero(dim(X)),
    for(j,1,dim(X),
    for(i,1,dim(X), test(sX[i]==X[j], M[j]=Rank(sX)[i]))), M)
--............................................................ END

-- PROBLEM: (cited from Zoefel [19, p.103])
-- As part of a biological study, butterflies were observed at various 
-- times within a fixed timeframe. 
-- The weather conditions were recorded, with a general 
-- classification into sunny and cloudy periods. 
-- The results are shown in the table:
--     sunny : 6 15 35 35 62 73 98 112
--     cloudy: 1  4  8 17 23 34 43

-- An U test should explain whether the difference between the two 
-- median values of the samples is significant.

-- SOLUTION:

"FIRST: We do a step-by-step solution."

X = (6, 15, 35, 35, 62, 73, 98, 112)
Y = (1, 4, 8, 17, 23, 34, 43)

median(X)  -- 48.5
median(Y)  -- 17

-- rank.ing the data by eye
TX    = zero(4, 11)
TX[1] = ("sun:", 6, 15, 35, 35, 62, 73, 98, 112, "", "sum:")
TX[2] = ("rank:",3,5,9.5,9.5,12,13,14,15,"___", "")
TX[3] = ("cloud:", 1, 4, 8, 17, 23, 34, 43,"", "", "")
TX[4] = ("rank:", 1,2,4,6,7,8,11,"","___", 0 )
TX[2,11] = sum(j,2,9, TX[2,j])
TX[4,11] = sum(j,2,8, TX[4,j]) 
transpose(TX)

-- We now create the process from column to column.
-- First collect data X and Y in one sample Z

Z= zero(dim(X)+dim(Y))       
   for(i,1,dim(X), Z[i]=X[i])
   for(i,1,dim(Y), Z[i+dim(X)]=Y[i])
Z 

Zs = sort(Z)    
Zs

-- establish a correspondence between Z and the sorted Zs 
-- which gives the rank numbers for each entry of Z

Zc = correspond(Z,Zs)         -- ok
Zc 

-- now add the ranks of entrys in X and Y by splitting Z -> X,Y

R1 = sum(i,1,dim(X), Zc[i])          -- rank sum of X
R1

R2 = sum(i,1,dim(Y), Zc[i+dim(X)])   -- rank sum of y
R2

-- now calculate the U statistic

n1 = dim(X)
n1
n2 = dim(Y)
n2

-- alternative determination of the ranks using 'eye' table TX
-- R1 =  sum(j,2,8, TX[4,j])   
-- R1
-- R2 =  sum(j,2,9, TX[2,j]) 
-- R2

n = n1+n2
n

U1 = R1 - n1*(n1+1)/2
U1

U2 = R2 - n2*(n2+1)/2
U2

-- critical value U to test is the minimum of U1 and U2

U = min(U1,U2)      
U

"--- 3. --- ranksum (MatLab) alias wilcox.test (R) in EIGENMATH"
-- function ranksum in Eigenmath to automate the above procedure.


################################################################
          ranksum(X,Y, Z,Zs,Zr, R1,R2,U1,U2) = 
################################################################
   do( -- we use pValue criterion instead of Ucritical
       mX = median(X),  
       mY = median(Y),  
       Z  = zero(dim(X)+dim(Y)), 
             for(i,1,dim(X), Z[i]=X[i]),
             for(i,1,dim(Y), Z[i+dim(X)]=Y[i]),
       Zs = sort(Z),      
       Zc = correspond(Z,Zs),       
       R1 = sum(i,1,dim(X), Zc[i]),         
       R2 = sum(i,1,dim(Y), Zc[i+dim(X)]),  
       n  = dim(X)+dim(Y),
       U1 = R1 - n1*(n1+1)/2,
       U2 = R2 - n2*(n2+1)/2,
       U  = min(U1,U2),      
       T  = (R2-0.5*n2*(n+1))/sqrt(n1*n2*(n+1)/12),
       pValue = stdnormCDF(mag(T)),
       print("  _______ ranksum _______"),
       Ho = test(pValue < 0.975, "accept H0", "reject H0"),
      (("U", "p", "Ho"), (U, pValue, Ho )))
########################################################### END

-- Check:

X = (6, 15, 35, 35, 62, 73, 98, 112)
Y = (1, 4, 8, 17, 23, 34, 43)
ranksum(X,Y)
